Just use a different rewrite if there are no arumgents:

function_call :
  ID '.' '(' 
	(
         argument_list ')'
 	       ->    <<
 	       template(function_name={$ID.text},
 
function_args={$argument_list.st})
                   >>
        | ')'
             -> << something else >>
      )
 	      ;
