// parser rule. this works.
expression
     : (a=term -> $a) ( ( '|' b=term -> ^(OR $expression $b)
                        | '|'        -> ^(OR $expression EPSILON)
                        )
                      )*
     ;

// tree grammar 1, based on CMinus example in the book.Throws a

     : IF_ACTION

     | ID ATTRIBUTES
     | STRINGLITERAL
     | EPSILON
     | ^(GROUPED_EXPR expression)
