This should do it

Alexander

start    :    (Float|UINT)+ EOF;

Float    :d=UINT
    (
           (DOT UINT)=>(DOT UINT)
           |{ $type = UINT; }
            )
            ;
           
DOT    :'.';           
fragment
DIGIT_10    :'0'..'9' ;
fragment
UINT    :DIGIT_10+ ;
