grammar Test;

x returns [List<Integer> values]
@init {
$values = new ArrayList<Integer>();
}
	:	e=y {$values.add($e.value)} (',' e=y {$values.add($e.value)} )*
	;

y returns [int value]
	: DIGIT+ { $value = Integer.parseInt($text); }
	;

DIGIT	:	'0'..'9'
	;
	
WS	:	(' ' | '\t' | '\n' | '\r') { $channel=HIDDEN; }
	;
