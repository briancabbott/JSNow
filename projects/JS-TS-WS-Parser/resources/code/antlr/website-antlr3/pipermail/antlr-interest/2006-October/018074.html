comando
               : comando_if
               |  comando_perform
               |  comando_call
...

comando_if
	: IF^ condicion
                       parte_then
                       (options {greedy=true;}: ELSE! parte_else)?
                  (options {greedy=true;}: END_IF)? 				;

parte_then
: (THEN!)?
       ( (NEXT^ SENTENCE {## = #( #[SENTENCIAS, "THEN"] ,##);})
          |sentencia_interna
       )
;
	
parte_else
: (NEXT^ SENTENCE {## = #( #[SENTENCIAS, "ELSE"] ,##);})	
| sentencia_interna
;
