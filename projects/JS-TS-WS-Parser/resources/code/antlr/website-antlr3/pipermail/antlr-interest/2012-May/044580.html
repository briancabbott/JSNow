Okay, I've tried to boil it down to a simple test case that illustrates
(probably in my parse rule) what I'm doing wrong. Target is C#.

 

//-----------------------------

// TreeTestInput.txt

Branch1
Branch2
Branch3

 

//-----------------------------

// Desired output:

ROOT

   Branch1

   Branch2

   Branch3

//-----------------------------

// Actual output:

ROOT

   Branch3

 

//-----------------------------

// TreeTest.g

grammar TreeTest;

 

options {

  language=CSharp3;

  output=AST;

}

 

tokens {

    ROOT;

}

 

public parse

    :  ( r=BRANCH )* EOF -> ^(ROOT $r*)

    ;

    

BRANCH

       :  'Branch' ('0'..'9')

       ;

              

EOL : ('\r'? '\n') {$channel=Hidden;} ; 

WS  : (' '|'\t'|'\f'| EOL)+ {$channel=Hidden;} ;
