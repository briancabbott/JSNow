Why does the below grammer successfully parse the input " B ==3C" When
there are no numerals allowed anywhere in the grammer?

grammar BoolExpr;

top	:
	expr EOF
	;

expr	: 
	unaryExp | equalityExp
	;

unaryExp:	
	'!' equalityExp
	;

equalityExp	:
	andOrExpr (('=='|'!=') andOrExpr)*
	;

andOrExpr
	:	atom (('||'|'&&') atom)*
	;

atom	:
	|	ID
	|	'(' expr ')'
	;

ID	:	('A'..'Z')+
	;

WS  	:  (' '|'\r'|'\t'|'\u000C'|'\n') {$channel=HIDDEN;}
    	;
