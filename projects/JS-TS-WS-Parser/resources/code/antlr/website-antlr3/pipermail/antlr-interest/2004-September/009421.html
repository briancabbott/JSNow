Something like this is one way.

class LTest extends Lexer;

tokens {
  FUNC;
}

VAR : ("plus" ('a'..'z'|'0'..'9')) => ('a'..'z') ('a'..'z'|'0'..'9')*
  | ("plus")=> "plus" {$setType(FUNC);};

WS : ( ' '| '\t' | '\f') { $setType(Token.SKIP); }
