don't know what to throw etc ...)?

filter=DEREMER

protected DEREMER [returns token] : (
	(newline|semi)
	(id:IDENT {if text != "REM" throw tokenmatchexception}|"*"|"!")

(newline | semi) ("rem" | "*" | "!" ) ~( "=" | "(" ) ({greedy=false} .)
newline)
