How about... (and note that I changed that to + rather than * as I don't
like rules that match nothing, you should use jText? instead :-)

jText returns [String text]
@init {$text = ""}
         :
		(       id=(STRING | EQUALS | PIPE | SPACE)
                 {
                     $text = $text + $id.text;
                 }
		)+
         ;
