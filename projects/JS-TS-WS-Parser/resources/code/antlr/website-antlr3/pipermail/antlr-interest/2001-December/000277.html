That generates this code:

		int counter=0
		
		try {      // for error handling
			counter=blockCount();
			{
			_loop3:
			do {
				if (((LA(1)==LITERAL_A))&&(counter>0)) {
					blockEntry();
					counter--;
				}
				else {
					break _loop3;
				}
				
			} while (true);
			}
		}
		...
