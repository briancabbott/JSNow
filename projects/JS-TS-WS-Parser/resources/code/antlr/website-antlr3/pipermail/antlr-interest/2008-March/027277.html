conditionalStatement
    : (KW_IF blah KW_THEN ID KW_ELSE ID) => (KW_IF^ blah KW_THEN! ID
KW_ELSE! ID)
    | (KW_IF blah KW_THEN ID) => (KW_IF^ blah KW_THEN! ID)
    ;


termExpression
    : binaryExpression *// ((STAR^ | SLASH^ | KW_DIV^ | KW_MOD^)
binaryExpression)**

binaryExpression

unaryExpression)?

unaryExpression
