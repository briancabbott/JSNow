// *************************** 
grammar test;
 
globalnumber : phonedigit (NEWLINE)* ;
 
phonedigit : (DIGIT)* ; 
 
NEWLINE : '\n' ;

ALPHA : ('a'-'z') | ('A'-'Z'); 
 

// ***************************
