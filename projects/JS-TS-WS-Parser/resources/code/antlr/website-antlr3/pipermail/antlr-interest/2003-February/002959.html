Try by adding a ‘+’ behind :

NEWLINE
	: ("\r\n" | '\r' | '\n')+
	{ newline(); $setType(Token.SKIP); }
;

WS
	: (' '|'\t')+
	{ $setType(Token.SKIP); }
;
