> I think this is what you want or at least really close.
>
> grammar Sample;
>
> start
>   options { backtrack = true; }
>   : (floatValue | integerValue) DOT IDENTIFIER
>     { System.out.println("matched!"); }
>   ;
>
> floatValue: NUMBER DOT NUMBER;
> integerValue: NUMBER;
>
> DOT: '.';
> IDENTIFIER: LETTER+;
> NUMBER: DIGIT+;
> fragment LETTER: 'a'..'z';
> fragment DIGIT: '0'..'9';
>
> NEWLINE: '\r'? '\n' { skip(); };
