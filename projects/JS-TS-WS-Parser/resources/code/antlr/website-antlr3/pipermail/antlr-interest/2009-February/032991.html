How about this...

NONCOMMENT  : '#ABCD' { $channel = HIDDEN; }
            ;

COMMENT     : '#' ~('\r' | '\n')* NEWLINE   { $channel = HIDDEN; }
            ;

NEWLINE     : '\r' ('\n')?
            | '\n'
            ;
