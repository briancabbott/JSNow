I move only the one, two... and keep the 0..9 in the lexer, of course.





> class TestParser extends Parser;
> options {
> 	k=1;
> }
>
> expr
> {int v;}
> 	: v = EN_NATURAL_NUMBER
> 	;
>
> class TestLexer extends Lexer;
> options {
> 	k=2;
> 	testLiterals=false;
> }
>
> protected EN_NATURAL_NUMBER returns [int v] :
>     | ('0'..'9')+ { v=Integer.valueOf($getText); }
>     | "one"     { v=1; }
>     | "two"     { v=2; }
>     | "three"   { v=3; }
>     | "four"    { v=4; }
>     | "five"    { v=5; }
>     | "six"     { v=6; }
>     | "seven"   { v=7; }
>     | "eight"   { v=8; }
>     | "nine"    { v=9; }
>     | "ten"     { v=10;}
>     ;
>
>
