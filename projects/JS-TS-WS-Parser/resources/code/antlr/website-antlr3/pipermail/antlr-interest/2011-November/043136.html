You don't want that anyway as your nodes will not have token start and
stop information and you might need that.

method_declaration
     : method_header method_body
     ;

method_body
	: (
		  ';'!	
     	| '{'!   statement_list?   '}'!
       )
		-> ^(METHOD_BODY statement_list?)
     ;

method_header
	:  attributes ID etc

		->^(METHOD_HEADER attributes ID )
	;
