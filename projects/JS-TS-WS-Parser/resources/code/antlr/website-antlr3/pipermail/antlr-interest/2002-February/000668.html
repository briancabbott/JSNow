Use a label.

 declaration
     { MyNode aNode = new MyNode(); }
     :
      #( decl:DeclarationNode
             declaration[aNode]
             statementBlock
        )
     { aNode.addASTNode(decl);
       someDataStructure.add(aNode);
     }
