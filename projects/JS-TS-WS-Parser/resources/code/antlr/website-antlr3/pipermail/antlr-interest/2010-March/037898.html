grammar JScriptDoc;

options
{
     output=AST;
     backtrack=true;
     memoize=true;
}

source
     : (statement)*
     ;

statement
     : statementEmpty
     | statementBlock
     ;

statementBlock

     : '{' (statement )* '}'
     ;

statementEmpty
     : ';'
     ;

OPERATOR_LT
     : '<'
     ;

OPERATOR_DOT
     : '.'
     ;

IDENTIFIER
     : ('a'..'z'|'A'..'Z') ('a'..'z'|'A'..'Z'|'0'..'9'|'_')*
      ;

NEWLINE
     : '\r'? '\n'
     {$channel=HIDDEN;}
         ;

WHITESPACE
     : ' '
     {$channel=HIDDEN;}
     ;

WHITETAB
     : '\t'
     {$channel=HIDDEN;}
     ;
