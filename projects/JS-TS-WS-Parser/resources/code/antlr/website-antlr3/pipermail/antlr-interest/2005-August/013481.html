public class LineNumberAST : CommonAST
{
	private int _line;
	private int _col;

	public LineNumberAST()
	{
	}

	public LineNumberAST(Token tok) : base(tok) 
	{
	}

	public override void initialize(Token tok) 
	{
		base.initialize(tok);
		_line = tok.getLine();
		_col = tok.getColumn();
	}

	public int Line
	{
		get { return _line;  }
		set { _line = value; }
	}

	public int Column
	{
		get { return _col;  }
		set { _col = value; }
	}
}
