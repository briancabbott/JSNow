grammar T;

options { output=AST; language=Java; }

tokens {
  PARAGRAPH;
  SENTENCE;
  NOUN;
  VERB;
  ADJECTIVE;
  THIS="this";
  IS="is";
  A="a";
  BAD="bad";
  EXAMPLE="example";
}

parse : THIS IS A BAD EXAMPLE -> ^( PARAGRAPH ^(SENTENCE ^(NOUN EXAMPLE)
^(ADJECTIVE BAD) ^(VERB IS)))
         ;
