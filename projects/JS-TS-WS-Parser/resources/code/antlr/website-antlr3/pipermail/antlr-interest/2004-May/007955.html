Basic idea:

method:          #(i:METHOD_ID
     { Array a; }   a=argList)
     { ...call i with a... }
     ;

argList
     returns [ Array a ]:
     { a = new Array; }
     ( { Value v; } v=expression { a.add(v); } )*
     ;
