This is the easiest way, though there are others such as ('""')=>:

DQ_LITERAL
   : '"' (options { greedy = false; } 
              : ( 
                   ( 
                       {input.LA(1) == '"' && input.LA(2) == '"'}? '"' '"' 
                     | ~'"' 
                   )*
                )
             )  
         '"'
    ;
