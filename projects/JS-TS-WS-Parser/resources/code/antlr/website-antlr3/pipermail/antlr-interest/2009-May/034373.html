maybe is this what you want?

in      :   cname*;
cname   :   '@' CHAR* ':' NL param*;
param   :   (CHAR|':')+ NL+;
CHAR    :   ~(NewLineChars|'@'|':')+ ;
NL      :   NewLineChars+;

fragment NewLineChars
    : '\n' 
    | '\r' 
    ;
