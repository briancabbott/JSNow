 If you need to print out the function call later, that also
gives you a production into which you can place the parens and comma:

    arglist:
        #( ARGLIST
            { cout << "("; }
            (
                expr
                (
                    // commas *between* expressions, not *after* expressions
                    { cout << ", "; }
                    expr
                )*
            )?
            { cout << ")"; }
        )
    ;
