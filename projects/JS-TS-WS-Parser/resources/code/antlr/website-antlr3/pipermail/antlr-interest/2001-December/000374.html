>
> class CalcParser extends Parser;
> options {
>     buildAST = true;   // uses CommonAST by default
> }
> formula: expr SEMI!
>     ;
> expr:   mexpr (PLUS^ mexpr)*
>     ;
> mexpr
>     :   atom (STAR^ atom)*
>     ;
> atom:   (INT|pexpr)
>     ;
> pexpr: LPAREN! expr RPAREN!
>     ;
>
> ..simple!
