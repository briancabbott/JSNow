I figured it out. It worked when I used multi-valued attributes.
  Thanks.
  --Sarim.
   
  group RuleTemplate;
fulldef(package,import,rulename) ::= <<$packagedef()$
$import:importdef()$
$rulestart()$
>>
packagedef()::="package $package$"
importdef()::=<<
import $it$;
   
  >>
rulestart()::="rule $rulename$ "
