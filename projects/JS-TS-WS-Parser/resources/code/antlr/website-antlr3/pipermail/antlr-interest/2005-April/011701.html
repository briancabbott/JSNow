There's several ways to specify a default for xy:

    /**
     * initialize in declaration
     */
    range_opt returns [pair<int,int> xy = ... ]
    :
        ( xy=range )?
    ;


    /**
     * initialize in preamble
     */
    range_opt returns [pair<int,int> xy]
    { xy = ... ; }
    :
        ( xy=range ) ?
    ;

    /**
     * initialize in alternative - note that
     * the ()? is replaced with ( | )
     */
    range_opt returns [pair<int,int> xy]
    :
        ( /* nothing here */
            { xy = ...; }
        | xy=range
        )
    ;
