>                    #(NS_LABEL #(PREFIX prefix:NCNAME)

>                      ## = #([OPERATORS,"operators"],
>                               #([TYPE_SEL,"type"], [STRING,"tree"]),

>                               #([LABEL_SEL,"label"], label),
