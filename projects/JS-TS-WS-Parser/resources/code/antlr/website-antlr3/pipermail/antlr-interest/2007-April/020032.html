Just in case someone wants to know how I solved (:


tableAlias
	: (schemaName DOT)? tableName (AT_SIGN linkName)? (aliasLabel=alias)?
	  {refs.notityTableRef($tableName.text, aliasLabel == null ? null :
$aliasLabel.text, this.scope);}
	;
