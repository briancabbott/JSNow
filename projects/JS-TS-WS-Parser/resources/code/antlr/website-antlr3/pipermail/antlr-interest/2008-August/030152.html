@parser::members
{
 static void MY_displayError(pANTLR3_BASE_RECOGNIZER _pRec,
pANTLR3_UINT8* _tokenNames) {
  //do something...
 }
}
@parser::apifuncs
{
 // Install custom error message display:
 RECOGNIZER->displayRecognitionError = MY_displayError;
