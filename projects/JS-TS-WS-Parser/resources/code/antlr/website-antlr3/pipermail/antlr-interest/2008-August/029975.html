Found a script to kill these pesky OS X turds:

for i in $(ls -Rl@ | grep '    ' | awk '{print $1}' | sort -u);    do  
echo echo Removing $i ... >&2;  find . -print0 | xargs -0t xattr -d $i  
2>/dev/null ; done
