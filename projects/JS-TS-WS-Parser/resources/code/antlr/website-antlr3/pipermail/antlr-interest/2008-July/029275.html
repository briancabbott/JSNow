This is a simplified version of the grammar I am working on.

grammar Test;

//comment out from_clause
//and warning in qualified_name
//goes away

       :       'FROM' qualified_name ;

column_reference
       :       (qualified_name PERIOD)? 'a';

       :       ('a' PERIOD)? 'a';

WS
   : (' '|'\t'|'\n'|'\r')+ {channel=HIDDEN;};
