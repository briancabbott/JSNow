subtract : modulo ('-' modulo)* -> ^(SUBTRACT modulo (modulo)* ) ;


subtract
    :
    modulo
    ( modul
       ( '-' modulo)+  -> ^(SUBTRACT modulo+) // you don't need the
parenthesis or the asterisk
       |
      () -> modulo
    )
   ;
