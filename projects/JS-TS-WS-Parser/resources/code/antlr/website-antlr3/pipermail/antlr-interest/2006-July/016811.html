function main
  define AAA integer
  define somename string
  AAA = 2
  somename = "BBBB"
  CCCC(AAA, somename)
end main

  if somename == "BABA"
    return AAA
  else
    return 0
end CCCC

where AAA, BBBB, CCCC and BABA are all chinese character strings.
